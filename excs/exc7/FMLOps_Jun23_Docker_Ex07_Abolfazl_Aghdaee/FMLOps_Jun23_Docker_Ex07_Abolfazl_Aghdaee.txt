version: '3.8'

services:
  flask_app:
    image: python:3.8
    container_name: flask_app
    working_dir: /app
    volumes:
      - .:/app
    ports:
      - "5000:5000"
    command: >
      sh -c "pip install flask && python -m flask run --host=0.0.0.0"

  redis:
    image: "redis:alpine"
    container_name: redis
    ports:
      - "6379:6379"
